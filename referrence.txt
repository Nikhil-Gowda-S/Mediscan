Import Torchvision
From Torchvision Import Models
Import Torch.Nn As Nn
Import Torch.Nn.Functional As F
From Transformers Import Bertmodel, Berttokenizer
Import Whisper
Import Librosa
Import Soundfile As Sf
Import Numpy As Np
From Pathlib Import Path

Class Modelloader:
    Def __init__(Self):
        Self.Device = Torch.Device("Cuda" If Torch.Cuda.Is_Available() Else "Cpu")
        Self.Image_Backbone = Self.Load_Image_Encoder()
        Self.Audio_Model = Self.Load_Audio_Encoder()
        Self.Nlp_Model = Self.Load_Nlp_Encoder()
        Self.Fusion_Classifier = Self.Load_Fusion_Classifier()
        
    Def Load_Image_Encoder(Self):
        """Dense-121 Pretrained On ImageNet, Adapted For X-Ray"""
        Model = Models.Densenet121(Pretrained=True)
        # Modify First Conv Layer For Grayscale X-Rays
        Model.Features.Conv0 = Nn.Conv2d(1, 64, Kernel_Size=7, Stride=2, Padding=3, Bias=False)
        # Modify Classifier For 14 Disease Classes
        Model.Classifier = Nn.Sequential(
            Nn.Linear(1024, 512),
            Nn.ReLU(),
            Nn.Dropout(0.3),
            Nn.Linear(512, 256)
        )
        Model = Model.To(Self.Device)
        Model = Model.Eval()
        Return Model
    
    Def Load_Audio_Encoder(Self):
        """Whisper For Speech-To-Text"""
        Model = Whisper.Load_Model("Base")
        Return Model.To(Self.Device)
    
    Def Load_Nlp_Encoder(Self):
        """Clinical BioBERT For Symptom Analysis"""
        Tokenizer = Berttokenizer.From_Pretrained("Emilyalsentzer/Clinical-Bert")
        Model = Bertmodel.From_Pretrained("Emilyalsentzer/Clinical-Bert").To(Self.Device)
        Return Tokenizer, Model
    
    Def Load_Fusion_Classifier(Self):
        """Fusion + Classification Head"""
        Class Fusionclassifier(Nn.Module):
            Def __init__(Self):
                Super().__init__()
                # Cross-Modal Attention
                Self.Image_Query = Nn.Linear(1024, 256)
                Self.Audio_Key = Nn.Linear(256, 256)
                Self.Audio_Value = Nn.Linear(256, 256)
                Self.Vital_Key = Nn.Linear(10, 64)
                Self.Vital_Value = Nn.Linear(10, 64)
                
                # Classification Head
                Self.Fusion = Nn.Sequential(
                    Nn.Linear(1024 + 256 + 64, 512),
                    Nn.LayerNorm(512),
                    Nn.ReLU(),
                    Nn.Dropout(0.3),
                    Nn.Linear(512, 256),
                    Nn.LayerNorm(256),
                    Nn.ReLU,
                    Nn.Dropout(0.3),
                    Nn.Linear(256, 14),  # 14 Disease Classes
                    Nn.Softmax(Dim=1)
                )
                
                # Uncertainty Quantification
                Self.Dropout = Nn.Dropout(0.2)
            
            Def Forward(Self, Image_Features, Audio_Features, Vital_Features):
                # Multi-Head Cross Attention
                Image_Q = Self.Image_Query(Image_Features).Unsqueeze(1)  # [B, 1, 256]
                Audio_K = Self.Audio_Key(Audio_Features).Unsqueeze(1)  # [B, 1, 256]
                Audio_V = Self.Audio_Value(Audio_Features).Unsqueeze(1)  # [B, 1, 256]
                
                # Cross Attention: Image -> Audio
                Audio_Attn = F.Softmax(Image_Q @ Audio_K.Transpose(-2, -1) / 16.0, Dim=-1)
                Audio_Context = Audio_Attn @ Audio_V  # [B, 1, 256]
                
                Vital_K = Self.Vital_Key(Vital_Features).Unsqueeze(1)
                Vital_V = Self.Vital_Value(Vital_Features).Unsqueeze(1)
                Vital_Attn = F.Softmax(Image_Q @ Vital_K.Transpose(-2, -1) / 8.0, Dim=-1)
                Vital_Context = Vital_Attn @ Vital_V  # [B, 1, 64]
                
                # Fusion + Classification
                Fused = Torch.Cat([
                    Image_Features,
                    Audio_Context.Squeeze(1),
                    Vital_Context.Squeeze(1)
                ], Dim=1)
                
                # Monte Carlo Dropout For Uncertainty
                Output = Self.Fusion(Fused)
                Output = Self.Dropout(Output)
                Output = Self.Fusion[-1](Output)  # Final Linear
                
                Return Output
            
        Return Fusionclassifier().To(Self.Device)

From Transformers Import Bertmodel, Berttokenizer

Class Audioencoder:
    Def __init__(Self, Model, Nlp_Model, Tokenizer):
        Self.Whisper = Model
        Self.Nlp = Nlp_Model
        Self.Tokenizer = Tokenizer
        Self.Projection = Nn.Sequential(
            Nn.Linear(768, 512),  # Whisper Dim -> 512
            Nn.ReLU(),
            Nn.Linear(512, 256)
        )
    
    Def Encode(Self, Audio_Path):
        # Speech To Text
        Result = Self.Whisper.Decode(Audio_Path)
        Text = Result["Text"]
        
        # Clinical Nlp Embeddings
        Inputs = Self.Tokenizer(Text, Return_Tensors="Pt", Padding=True, Truncation=True, Max_Length=128)
        With Torch.No_Grad():
            Outputs = Self.Nlp(**Inputs)
            Embeddings = Outputs.Last_Hidden_State[:, 0, :]  # [Cls] Token
        
        # Project To 256-D
        Embeddings = Self.Projection(Embeddings)
        Return Embeddings.Squeeze(1)

#### **Models/Fusion_Classifier.Py**
```Python
Import Torch
Import Torch.Nn As Nn
Import Torch.Nn.Functional As F

class FusionClassifier(nn.Module):
    def __init__(self):
        super().__init__()
        # Cross-modal attention layers
        self.image_query = nn.Linear(1024, 256)
        self.audio_key = nn.Linear(256, 256)
        self.audio_value = nn.Linear(256, 256)
        self.vital_key = nn.Linear(10, 64)
        self.vital_value = nn.Linear(10, 64)
        
        # Classification head
        self.fusion = nn.Sequential(
            nn.Linear(1024 + 256 + 64, 512),
            nn.LayerNorm(512),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(512, 256),
            nn.LayerNorm(256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, 14),
            nn.Softmax(dim=1)
        )
        
        # Uncertainty quantification
        self.dropout = nn.Dropout(0.2)
    
    def forward(self, image_features, audio_features, vital_features):
        # Cross-attention: Image queries Audio and Vitals
        image_q = self.image_query(image_features).unsqueeze(1)  # [B, 1, 256]
        audio_k = self.audio_key(audio_features).unsqueeze(1)  # [B, 1, 256]
        audio_v = self.audio_value(audio_features).unsqueeze(1)  # [B, 1, 256]
        
        # Cross attention: Image -> Audio
        audio_attn = F.softmax(image_q @ audio_k.transpose(-2, -1) / 16.0, dim=-1)
        audio_context = audio_attn @ audio_v  # [B, 1, 256]
        
        vital_k = self.vital_key(vital_features).unsqueeze(1)
        vital_v = self.vital_value(vital_features).unsqueeze(1)
        vital_attn = F.softmax(image_q @ vital_k.transpose(-2, -1) / 8.0, dim=-1)
        vital_context = vital_attn @ vital_v  # [B, 1, 64]
        
        # Fusion + Classification
        fused = torch.cat([
            image_features,
            audio_context.squeeze(1),
            vital_context.squeeze(1)
        ], dim=1)
        
        # Monte Carlo Dropout for uncertainty
        output = self.fusion(fused)
        output = self.dropout(output)
        output = self.fusion[-1](output)  # Final linear
        
        return output
```

#### **App/Components/Explainability.Py**
```Python
Import Cv2
Import Numpy As Np
From PIL Import Image
Import Base64
From Io Import Bytesio

Class Gradcamoverlay:
    Def Create_Overlay(Self, Original_Image, Attributions, Alpha=0.5):
        """Create Grad-CAM heatmap overlay"""
        Original = Np.Array(Original_Image)
        Attributions = Attributions.Detach().Cpu().Numpy()
        
        # Normalize attributions
        Attributions = Np.Maximum(Attributions, 0)
        Attributions = Attributions / Np.Max(Attributions)
        
        # Create Heatmap
        Heatmap = Cv2.Applycolormap(
            Np.Uint8(255 * Attributions), 
            Cv2.Color_Map_Jet
        )
        Heatmap = Cv2.Addweighted(
            Original[..., ::-1],  # Rgb
            Alpha, 
            Heatmap, 
            1 - Alpha, 
            0
        )
        
        Return Heatmap
```
